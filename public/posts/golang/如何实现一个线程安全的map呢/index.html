Golang 的 Map 不但是 无序 的，而且还是 非线程安全的。

读写锁 实现线程安全 amp 🔗

定义 🔗

type SafeMap struct {
	mu sync.RWMutex
	m  map[string]int
}

func NewSafeMap() *SafeMap {
	return &SafeMap{
		m: make(map[string]int),
	}
}

func (sm *SafeMap) Get(key string) (int, bool) {
	sm.mu.RLock()
	defer sm.mu.RUnlock()
	val, ok := sm.m[key]
	return val, ok
}

func (sm *SafeMap) Set(key string, value int) {
	sm.mu.Lock()
	defer sm.mu.Unlock()
	sm.m[key] = value
}

使用 🔗

safeMap := NewSafeMap()

// 读操作
val, ok := safeMap.Get("key")
if ok {
	println("Read:", val)
}

// 写操作
safeMap.Set("key", 1)
